    <script>
        // Custom Cursor
        const cursor = document.getElementById('cursor');
        document.addEventListener('mousemove', (e) => {
            gsap.to(cursor, {
                x: e.clientX - 20,
                y: e.clientY - 20,
                duration: 0.1
            });
        });

        // Register GSAP ScrollTrigger
        gsap.registerPlugin(ScrollTrigger);

        // Skew effect on scroll
        let proxy = { skew: 0 },
            skewSetter = gsap.quickSetter(".skew-box", "skewY", "deg"),
            clamp = gsap.utils.clamp(-20, 20); 

        ScrollTrigger.create({
            onUpdate: (self) => {
                let skew = clamp(self.getVelocity() / -300);
                if (Math.abs(skew) > Math.abs(proxy.skew)) {
                    proxy.skew = skew;
                    gsap.to(proxy, {skew: 0, duration: 0.8, ease: "power3", overwrite: true, onUpdate: () => skewSetter(proxy.skew)});
                }
            }
        });

        // Hero text parallax
        gsap.to("#hero-text", {
            scrollTrigger: {
                trigger: "body",
                start: "top top",
                end: "bottom top",
                scrub: true
            },
            y: 200,
            scale: 1.1,
            rotation: 5
        });

        // Make floating images move slightly on scroll
        gsap.to(".floating", {
            scrollTrigger: {
                scrub: 1
            },
            y: -150,
            rotation: -10
        });
    </script>
